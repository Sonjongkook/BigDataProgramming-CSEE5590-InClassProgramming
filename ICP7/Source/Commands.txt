$ sudo apt-get install Cassandra
$ sudo service cassandra start

$ create keyspace data with replication={'class':'SimpleStrategy', 'replication_factor':3};

$ use data;
$ create table data.employees (employee_id int PRIMARY KEY, department text, lastname text, year_with_company int, hiredate text, jobtitle text, salary int, managerid int);
$ copy employees (employee_id, department, lastname, year_with_company, hiredate, jobtitle, salary, managerid) from '/home/jk/Downloads/employee_entries.csv' with DELIMITER='|' AND HEADER=TRUE;
$ select * from employees;

Query1
$ select employee_id, lastname, jobtitle, hiredate from employees;
Query2
$ select lastname, salary from employees where jobtitle = 'clerk' ALLOW FILTERING;
Query3
$ select lastname, jobtitle, salary from employees where hiredate='2000-02-18' ALLOW FILTERING;
Query4
$ select lastname, salary from employees;
Query5
$ select lastname, salary, managerid from employees where salary=45000 ALLOW FILTERING;
